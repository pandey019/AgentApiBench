    # Reset environment
    try:
        reset_resp = await env_client.post("/reset", json={"episode_id": task_id})
        reset_resp.raise_for_status()
    except Exception as e:
        # Fallback if the strict openenv-core format is fighting the kwargs
        reset_resp = await env_client.post("/reset", json={"data": {"task_id": task_id}})
        
    result = reset_resp.json()
